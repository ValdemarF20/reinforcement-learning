\section{The kiosk (\texttt{kiosk.py})}
\subsubsection*{{\color{red}Problem 1:  A basic blaster-business}}

\begin{align}
	N & = 14 \\
	\mbox{for $k=0,\dots,N$: }\quad	\mathcal{S}_k & = \{0, 1, \ldots, 20\} \\
	\mbox{for $k=0,\dots,N-1$: }\quad \mathcal{A}_k(x_k) & = \{0, 1, \ldots, 15\} \\
    \mbox{for $k=0,\dots,N-1$: }\quad f_k(x_k, u_k, w_k) & = \min\!\bigl(20,\,\max(0,\, x_k + u_k - w_k)\bigr) \\
	\mbox{for $k=0,\dots,N-1$: }\quad g_k(x_k, u_k, w_k) & = 1.5\,u_k - 2.1\,\min(x_k+u_k,\,w_k) \\
	g_N(x_N) & = 0 \\
    \mbox{for $k=0,\dots,N-1$: }\quad P_W(w_k | x_k, u_k) & =
	\begin{cases}
		0.3 & w_k = 0 \\
		0.6 & w_k = 3 \\
		0.1 & w_k = 6
	\end{cases}
\end{align}
Here $x_k \in \mathcal{S}_k$ is the inventory level at the start of day $k$, $u_k$ is the number of blasters ordered, and $w_k$ is the random customer demand (Tusken-raider distribution). The stage cost charges $c_o = 1.5$ credits per blaster ordered and earns $s_p = 2.1$ credits per blaster sold. Inventory is hard-capped at $n_s = 20$; the action set does not depend on $x_k$ or $k$.

\subsubsection*{{\color{red}Problem 2:  Warmup}}

The state space at $k=0$ returned by \texttt{warmup\_states()} is $\mathcal{S}_0 = \{0, 1, \ldots, 20\}$, so $|\mathcal{S}_0| = 21$. The action set returned by \texttt{warmup\_actions()} is $\mathcal{A}_0(x_0) = \{0, 1, \ldots, 15\}$, so $|\mathcal{A}_0(x_0)| = 16$.

\subsubsection*{{\color{red}Problem 3:  Manually computing $J_{N-1}$}}
		
	The last step of the DP algorithm gives:
	$$
	J_{N-1}(20)  = \min_{u_{N-1} } \EE_{w_{N-1} }\left[ g_{N-1}(20, u_{N-1}, w_{N-1} ) +  g_N(x_N) \right]
	$$
	Since $g_N = 0$ and $x_{N-1} = 20 \ge 6 \ge w$ for all $w \in \{0,3,6\}$, we have $\min(20+u,w) = w$ for any $u \ge 0$. Therefore:
	\begin{align*}
		\EE\!\left[g_{N-1}(20,u,w)\right]
		&= 1.5\,u - 2.1\,\EE[w] \\
		&= 1.5\,u - 2.1\,(0.3\cdot 0 + 0.6\cdot 3 + 0.1\cdot 6) \\
		&= 1.5\,u - 2.1 \times 2.4 \;=\; 1.5\,u - 5.04
	\end{align*}
	This expression is strictly increasing in $u$ (coefficient $1.5 > 0$), so the minimiser is $u^* = 0$, giving:
	$$J_{N-1}(20) = -5.04, \qquad \mu_{N-1}(20) = 0.$$

\subsubsection*{{\color{red}Problem 6:  Policy explanation continued}}
	
	$$\mu_{N-1}(0) = \argmin_u \EE_{w} \left[g_k(0, u, w) + g_N( f(0,u,w) ) \right] = 0 $$

	At state $x=0$ on the last day with $g_N = 0$:
	\begin{align*}
		\EE\!\left[g_{N-1}(0,u,w)\right]
		&= 1.5\,u - 2.1\,\EE[\min(u,w)] \\
		&= 1.5\,u - 2.1\,(0.3\cdot\min(u,0) + 0.6\cdot\min(u,3) + 0.1\cdot\min(u,6))
	\end{align*}
	For $u \le 3$: all $\min(u,\cdot)$ terms equal $u$, so $\EE[g] = 1.5u - 2.1(0.6u+0.1u) = 0.03u \ge 0$, which is increasing in $u$. The minimiser is therefore $\mu_{N-1}(0) = 0$. Economically: on the last day with empty stock, the marginal ordering cost (1.5) exceeds the expected marginal revenue $2.1 \times (0.6+0.1) = 1.47$.

\section{Avoid the droid (\texttt{pacman.py})}

\subsubsection*{{\color{red}Problem 7:  Go east}}

\texttt{go\_east} initialises a \texttt{PacmanEnvironment} with the given map string, resets it to obtain the initial \texttt{GameState} $s_0$, then repeatedly calls \texttt{x.f('East')} until \texttt{x.is\_won()} is \texttt{True}. For the \texttt{east} map, 4 East-actions are needed, so the returned list is $[s_0, s_1, s_2, s_3, s_4]$ where $s_4$ is the won state.

\subsubsection*{{\color{red}Problem 8:  Predict consequence of actions}}

With no ghosts the transition is deterministic: action $u$ in state $x$ leads to exactly one successor $x' = x.\texttt{f}(u)$ with probability 1. Consequently \texttt{p\_next}$(x, u)$ returns the single-entry dictionary $\{x' : 1.0\}$.

\subsubsection*{{\color{red}Problem 9:  Possible future states}}

\texttt{get\_future\_states} builds the reachable sets iteratively: $\mathcal{S}_0 = \{x_0\}$ and $\mathcal{S}_{k+1}$ collects all states reachable from $\mathcal{S}_k$ under all actions and all ghost randomness. For the \texttt{SS0tiny} map (no ghosts):
$$|\mathcal{S}_1| = 4, \qquad |\mathcal{S}_{10}| = 49.$$

\subsubsection*{{\color{red}Problem 10:  Shortest path}}

Using \texttt{shortest\_path} (step cost 1, terminal cost $10^9$ if not won) the optimal action sequences are:
\begin{itemize}
	\item \texttt{east}: \texttt{[East, East, East, East]}
	\item \texttt{datadiscs}: \texttt{[West, South, East, East, East, East, North, North]}
	\item \texttt{SS0tiny}: \texttt{[West, South, East, East, East]}
\end{itemize}

\subsubsection*{{\color{red}Problem 11:  Predict consequence of actions with one ghost}}

After Pac-Man moves deterministically, the ghost selects uniformly at random from its $m$ available actions. Each resulting next state therefore receives probability $1/m$. In \texttt{SS1tiny} the ghost has $m=3$ available actions, so \texttt{p\_next}$(x, \texttt{West})$ returns 3 entries each with probability $\tfrac{1}{3}$.

\subsubsection*{{\color{red}Problem 12:  Possible future states with one ghost}}

For the \texttt{SS1tiny} map (one ghost):
$$|\mathcal{S}_1| = 12.$$

\subsubsection*{{\color{red}Problem 13:  Optimal one-ghost planning}}

Using \texttt{win\_probability} with \texttt{PacmanDPModel} (terminal reward $-1$ for winning, 0 otherwise) and planning horizon $N=12$:
$$\text{win\_probability}(\texttt{SS1tiny},\; N=12) = 1.0.$$
With $N=5$ the win probability is already $0.5$ and reaches $1.0$ at $N=10$, showing that a sufficiently long horizon guarantees a winning strategy.

\subsubsection*{{\color{red}Problem 14:  Predict consequence of actions with several ghosts}}

With $G$ ghosts each moving independently and uniformly at random, the joint outcome probability is:
$$P(a_1,\ldots,a_G) = \prod_{i=1}^{G} \frac{1}{|\mathcal{A}_i|},$$
where $\mathcal{A}_i$ is the action set of ghost $i$. If distinct joint actions lead to the same state, their probabilities are accumulated. In \texttt{SS2tiny} (two ghosts, $|\mathcal{A}_i|=3$ each) there are $3\times3=9$ equally probable joint outcomes, each with probability $\tfrac{1}{9}$, yielding 9 distinct successor states.

\subsubsection*{{\color{red}Problem 15:  Future states}}

For the \texttt{SS2tiny} map (two ghosts):
$$|\mathcal{S}_1| = 26.$$

\subsubsection*{{\color{red}Problem 16:  Optimal planning}}

Using \texttt{win\_probability} for \texttt{SS2tiny} with $N=12$:
$$\text{win\_probability}(\texttt{SS2tiny},\; N=12) \approx 0.495.$$
With two ghosts, even the optimal policy cannot guarantee a win; the win probability converges to $\approx 49.5\%$ as the horizon grows, reflecting the irreducible randomness of two independently moving ghosts.
